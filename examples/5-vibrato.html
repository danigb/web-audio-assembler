Web Audio Assembler Examples

5. Vibrato


var Synth = Assembler({
	amp: { node: 'Gain', gain: 0.4, connect: '$context' },
	filter: { node: 'BiquadFilter', type: 'lowpass', frequency: 400, connect: 'amp' },
	osc: { node: 'Oscillator', type: 'sawtooth', frequency: 880, connect: 'filter' },
	lfoGain: { node: 'Gain', gain: 10, connect: 'osc.frequency' },
	lfo: { node: 'Oscillator', frequency: 4, connect: 'lfoGain' }
})
Assembler.start(Synth(new AudioContext()))
