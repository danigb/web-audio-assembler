Web Audio Assembler Examples

2. Create simple synth


var ac = new AudioContext()
var Synth = Assembler({
	amp: ['Gain', { type: 'sawtooth', gain: 0.4, connect: 'destination' }],
	filter: ['BiquadFilter', { type: 'lowpass', frequency: 400, connect: 'amp' }],
	osc: [ 'Oscillator', { type: 'square', frequency: 700, connect: 'filter' }]
})
Synth(ac).osc.start()
